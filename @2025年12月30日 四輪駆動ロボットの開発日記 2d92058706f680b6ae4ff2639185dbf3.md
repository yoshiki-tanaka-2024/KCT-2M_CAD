### 3.3.2　ソースコード

- こちらをクリック
    
    ```cpp
    #include <M5Stack.h>
    #include <Wire.h>
    #include "Adafruit_TCS34725.h"
    
    // ===== TCA9548A =====
    #define TCA_ADDR 0x70
    
    void selectI2C(uint8_t ch) {
      if (ch > 7) return;
      Wire.beginTransmission(TCA_ADDR);
      Wire.write(1 << ch);
      Wire.endTransmission();
    }
    
    // ===== TCS34725 =====
    Adafruit_TCS34725 tcs1(
      TCS34725_INTEGRATIONTIME_50MS,
      TCS34725_GAIN_4X
    );
    
    Adafruit_TCS34725 tcs2(
      TCS34725_INTEGRATIONTIME_50MS,
      TCS34725_GAIN_4X
    );
    
    // ===== 初期描画 =====
    void drawStaticUI() {
      M5.Lcd.clear(BLACK);
      M5.Lcd.setTextSize(2);
    
      M5.Lcd.setCursor(0, 0);
      M5.Lcd.println("Sensor 1");
    
      M5.Lcd.setCursor(0, 120);
      M5.Lcd.println("Sensor 2");
    }
    
    // ===== センサ読み取り＆表示 =====
    void readAndDisplay(uint8_t ch, Adafruit_TCS34725 &tcs, int y) {
      uint16_t r, g, b, c;
    
      selectI2C(ch);
    
      // ★ Interrupt制御を使わない
      delay(60);  // 積分時間待ち
      tcs.getRawData(&r, &g, &b, &c);
    
      M5.Lcd.fillRect(0, y + 20, 320, 80, BLACK);
      M5.Lcd.setCursor(0, y + 20);
    
      M5.Lcd.printf("C:%4d\n", c);
      M5.Lcd.printf("R:%4d G:%4d B:%4d\n", r, g, b);
    
      String color = "UNKNOWN";
      if (c < 100) color = "DARK";
      else if (r > g && r > b) color = "RED";
      else if (g > r && g > b) color = "GREEN";
      else if (b > r && b > g) color = "BLUE";
    
      M5.Lcd.printf("Color: %s", color.c_str());
    }
    
    // ===== setup =====
    void setup() { 
      M5.begin(); 
      Wire.begin(21, 22); 
      drawStaticUI(); 
      
      // センサ1初期化 
      selectI2C(0);
      if (!tcs1.begin()) {
        M5.Lcd.println("Sensor1 ERROR");
      }
        
      // Sensor2
      selectI2C(1);
      if (!tcs2.begin()) {
        M5.Lcd.println("Sensor2 ERROR");
      }
    }
    
    // ===== loop =====
    void loop() {
      readAndDisplay(0, tcs1, 0);   // Sensor1
      readAndDisplay(1, tcs2, 120); // Sensor2
    
      delay(200);
    }
    
    ```

### 3.5.1　配線図

![image.png](image%203.png)

- M5StackBasicで使用できるPWMピンの数がおそらく不足している（モータを4つ動かす場合，PWMの信号線が8本必要になる）ため，PCA9685を用いて信号線を増設することでモータドライバへのPWM信号を送る方法をとる

### 3.5.2　ソースコード（M5Stack-PCA9685-MX1508で構成した場合）

- こちらをクリック
    
    ```c
    #include <Wire.h>
    #include <Adafruit_PWMServoDriver.h>
    
    Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver(0x40);
    
    #define MOTOR_A_IN1 0
    #define MOTOR_A_IN2 1
    #define MOTOR_B_IN1 2
    #define MOTOR_B_IN2 3
    
    void setup() {
      Wire.begin(21, 22);
      pwm.begin();
      pwm.setPWMFreq(1000);  // DCモータ向けPWM周波数（1kHz）
    }
    
    void loop() {
      motorA(4095/2);   // 正転
      motorB(4095/2);   // 正転
      delay(5000);
      motorA(0);
      motorB(0);
      delay(3000);
      motorA(-4095/2);  // 逆転
      motorB(-4095/2);  // 逆転
      delay(5000);
      motorA(0);      // 停止
      motorB(0);      // 停止
      delay(3000);
    }
    
    void motorA(int speed) {
      speed = constrain(speed, -4095, 4095);
    
      if (speed > 0) {
        pwm.setPWM(MOTOR_A_IN1, 0, speed);
        pwm.setPWM(MOTOR_A_IN2, 0, 0);
      } else if (speed < 0) {
        pwm.setPWM(MOTOR_A_IN1, 0, 0);
        pwm.setPWM(MOTOR_A_IN2, 0, -speed);
      } else {
        pwm.setPWM(MOTOR_A_IN1, 0, 0);
        pwm.setPWM(MOTOR_A_IN2, 0, 0);
      }
    }
    
    void motorB(int speed) {
      speed = constrain(speed, -4095, 4095);
    
      if (speed > 0) {
        pwm.setPWM(MOTOR_B_IN1, 0, speed);
        pwm.setPWM(MOTOR_B_IN2, 0, 0);
      } else if (speed < 0) {
        pwm.setPWM(MOTOR_B_IN1, 0, 0);
        pwm.setPWM(MOTOR_B_IN2, 0, -speed);
      } else {
        pwm.setPWM(MOTOR_B_IN1, 0, 0);
        pwm.setPWM(MOTOR_B_IN2, 0, 0);
      }
    }
    
    ```

### 3.5.4　ソースコード（M5Stack-TCA9548A-PCA9685-MX1508で構成した場合）

- こちらをクリック
    
    ```cpp
    #include <M5Stack.h>
    #include <Wire.h>
    #include <Adafruit_PWMServoDriver.h>
    
    // ================================
    // I2C Address
    // ================================
    #define TCA_ADDR 0x70
    #define PCA_ADDR 0x40
    #define PCA_CH   2   // PCA9685が接続されているTCAチャンネル
    
    // ================================
    // PCA9685
    // ================================
    Adafruit_PWMServoDriver pwm(PCA_ADDR);
    
    // MX1508 (PCA9685 PWMチャンネル)
    #define MOTOR_A_IN1 0
    #define MOTOR_A_IN2 1
    #define MOTOR_B_IN1 2
    #define MOTOR_B_IN2 3
    
    bool pca_ok = false;
    
    // ================================
    // TCA9548A チャンネル選択
    // ================================
    void selectTCA(uint8_t ch) {
      Wire.beginTransmission(TCA_ADDR);
      Wire.write(1 << ch);
      Wire.endTransmission();
    }
    
    // ================================
    // I2C ACK確認
    // ================================
    bool checkI2C(uint8_t addr) {
      Wire.beginTransmission(addr);
      return (Wire.endTransmission() == 0);
    }
    
    // ================================
    // モータ制御
    // speed: -4095 ～ +4095
    // ================================
    void motor(int in1, int in2, int speed) {
      speed = constrain(speed, -4095, 4095);
    
      if (speed > 0) {
        pwm.setPWM(in1, 0, speed);
        pwm.setPWM(in2, 0, 0);
      } else if (speed < 0) {
        pwm.setPWM(in1, 0, 0);
        pwm.setPWM(in2, 0, -speed);
      } else {
        pwm.setPWM(in1, 0, 0);
        pwm.setPWM(in2, 0, 0);
      }
    }
    
    // ================================
    // I2Cスキャン＆初期化
    // ================================
    void scanAndInit() {
      M5.Lcd.fillRect(0, 0, 320, 120, BLACK);
      M5.Lcd.setCursor(0, 0);
      M5.Lcd.println("[I2C STATUS]");
    
      // ---- TCA9548A ----
      bool tca_ok = checkI2C(TCA_ADDR);
      M5.Lcd.printf("TCA9548A : %s\n", tca_ok ? "OK" : "NG");
    
      pca_ok = false;
    
      // ---- PCA9685 ----
      if (tca_ok) {
        selectTCA(PCA_CH);
        pca_ok = checkI2C(PCA_ADDR);
    
        if (pca_ok) {
          pwm.begin();
          pwm.setPWMFreq(1000);   // DCモータ向け
        }
      }
    
      M5.Lcd.printf("PCA9685  : %s\n", pca_ok ? "OK" : "NG");
    
      M5.Lcd.println();
      M5.Lcd.println("[BUTTON]");
      M5.Lcd.println("A: Rescan I2C");
    }
    
    // ================================
    // setup
    // ================================
    void setup() {
      M5.begin();
      Wire.begin(21, 22);
    
      M5.Lcd.clear(BLACK);
      M5.Lcd.setTextSize(2);
    
      scanAndInit();
    }
    
    // ================================
    // loop
    // ================================
    void loop() {
      M5.update();
    
      // ---- Aボタン：I2C再スキャン ----
      if (M5.BtnA.wasPressed()) {
        M5.Lcd.fillRect(0, 140, 320, 100, BLACK);
        M5.Lcd.setCursor(0, 140);
        M5.Lcd.println("Rescanning...");
        delay(300);
        scanAndInit();
        return;
      }
    
      // PCAが見えていなければ何もしない
      if (!pca_ok) {
        delay(200);
        return;
      }
    
      // ---- モータ動作デモ ----
      M5.Lcd.fillRect(0, 140, 320, 100, BLACK);
      M5.Lcd.setCursor(0, 140);
      M5.Lcd.println("[MOTOR]");
      M5.Lcd.println("FORWARD");
    
      motor(MOTOR_A_IN1, MOTOR_A_IN2, 2048);
      motor(MOTOR_B_IN1, MOTOR_B_IN2, 2048);
      delay(3000);
    
      M5.Lcd.fillRect(0, 140, 320, 100, BLACK);
      M5.Lcd.setCursor(0, 140);
      M5.Lcd.println("[MOTOR]");
      M5.Lcd.println("STOP");
    
      motor(MOTOR_A_IN1, MOTOR_A_IN2, 0);
      motor(MOTOR_B_IN1, MOTOR_B_IN2, 0);
      delay(2000);
    
      M5.Lcd.fillRect(0, 140, 320, 100, BLACK);
      M5.Lcd.setCursor(0, 140);
      M5.Lcd.println("[MOTOR]");
      M5.Lcd.println("REVERSE");
    
      motor(MOTOR_A_IN1, MOTOR_A_IN2, -2048);
      motor(MOTOR_B_IN1, MOTOR_B_IN2, -2048);
      delay(3000);
    }
    
    ```
